165  t8_grid_element_to_vtk_cell (const grid_t grid, const t8_element_t *element, const t8_locidx_t itree,
166  const t8_gloidx_t offset, const int is_ghost, const int elem_id, long int *point_id,

172  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_element_id)
173  {
174  vtkSmartPointer<vtkCell> pvtkCell = NULL;
175 
176  /* Get the shape of the current element and the respective shape of the vtk_cell. */
177  const t8_element_shape_t element_shape = grid_element_shape (grid, itree, element);
178  const int num_node = t8_get_number_of_vtk_nodes (element_shape, curved_flag);
179 
180  if (curved_flag == 0) {
181  switch (element_shape) {
182  case T8_ECLASS_VERTEX:
183  pvtkCell = vtkSmartPointer<vtkVertex>::New ();
184  break;
185  case T8_ECLASS_LINE:
186  pvtkCell = vtkSmartPointer<vtkLine>::New ();
187  break;
188  case T8_ECLASS_QUAD:
189  pvtkCell = vtkSmartPointer<vtkQuad>::New ();
190  break;
191  case T8_ECLASS_TRIANGLE:
192  pvtkCell = vtkSmartPointer<vtkTriangle>::New ();
193  break;
194  case T8_ECLASS_HEX:
195  pvtkCell = vtkSmartPointer<vtkHexahedron>::New ();
196  break;
197  case T8_ECLASS_TET:
198  pvtkCell = vtkSmartPointer<vtkTetra>::New ();
199  break;
200  case T8_ECLASS_PRISM:
201  pvtkCell = vtkSmartPointer<vtkWedge>::New ();
202  break;
203  case T8_ECLASS_PYRAMID:
204  pvtkCell = vtkSmartPointer<vtkPyramid>::New ();
205  break;
206  default:
207  SC_ABORT_NOT_REACHED ();
208  }
209  }
210  else { /* curved_flag != 0 */
211  switch (element_shape) {
212  case T8_ECLASS_VERTEX:
213  pvtkCell = vtkSmartPointer<vtkVertex>::New ();
214  break;
215  case T8_ECLASS_LINE:
216  pvtkCell = vtkSmartPointer<vtkQuadraticEdge>::New ();
217  break;
218  case T8_ECLASS_QUAD:
219  pvtkCell = vtkSmartPointer<vtkQuadraticQuad>::New ();
220  break;
221  case T8_ECLASS_TRIANGLE:
222  pvtkCell = vtkSmartPointer<vtkQuadraticTriangle>::New ();
223  break;
224  case T8_ECLASS_HEX:
225  pvtkCell = vtkSmartPointer<vtkQuadraticHexahedron>::New ();
226  break;
227  case T8_ECLASS_TET:
228  pvtkCell = vtkSmartPointer<vtkQuadraticTetra>::New ();
229  break;
230  case T8_ECLASS_PRISM:
231  pvtkCell = vtkSmartPointer<vtkQuadraticWedge>::New ();
232  break;
233  case T8_ECLASS_PYRAMID:
234  pvtkCell = vtkSmartPointer<vtkQuadraticPyramid>::New ();
235  break;
236  default:
237  SC_ABORT_NOT_REACHED ();
238  }
239  }
240 
241  /* Compute the coordinates of the element/tree. */
242  double *coordinates = T8_ALLOC (double, 3 * num_node);
243 
244  grid_element_to_coords (grid, itree, element, curved_flag, coordinates, num_node, element_shape);
245 
246  for (int ivertex = 0; ivertex < num_node; ivertex++, (*point_id)++) {
247  const size_t offset_3d = 3 * ivertex;
248  /* Insert the point in the points array. */
249  points->InsertNextPoint (coordinates[offset_3d], coordinates[offset_3d + 1], coordinates[offset_3d + 2]);
250  pvtkCell->GetPointIds ()->SetId (ivertex, *point_id);
251  }
252  T8_FREE (coordinates);
253 
254  /* Fill the cell array. */
255  cellArray->InsertNextCell (pvtkCell);
256 
257  /* Write additional information if desired. */
258  if (curved_flag == 0) {
259  cellTypes[elem_id - offset] = t8_eclass_vtk_type[element_shape];
260  }
261  else {
262  cellTypes[elem_id - offset] = t8_curved_eclass_vtk_type[element_shape];
263  }
264  if (write_treeid == 1) {
265  const t8_gloidx_t gtree_id = tree_local_to_global_id (grid, itree);
266  if (is_ghost) {
267  vtk_treeid->InsertNextValue (-1);
268  }
269  else {
270  vtk_treeid->InsertNextValue (gtree_id);
271  }
272  }
273  if (write_mpirank == 1) {
274  int mpirank;
275  int mpiret = sc_MPI_Comm_rank (this->comm, &mpirank);
276  SC_CHECK_MPI (mpiret);
277  vtk_mpirank->InsertNextValue (mpirank);
278  }
279  if (write_level == 1) {
280  vtk_level->InsertNextValue (grid_element_level (grid, itree, element));
281  }
282  if (write_element_id == 1) {
283  vtk_element_id->InsertNextValue (elem_id);
284  }

323  void
324  t8_grid_to_vtkUnstructuredGrid (const grid_t grid, vtkSmartPointer<vtkUnstructuredGrid> unstructuredGrid)
325  {
326  T8_ASSERT (grid != NULL);
327 
328  vtkSmartPointer<vtkCellArray> cellArray = vtkSmartPointer<vtkCellArray>::New ();
329  vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New ();
330  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_treeid = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
331  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_mpirank = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
332  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_level = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
333  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_element_id = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
334 
335  /*
336  * We need the dataArray for writing double valued user defined data in the vtu files.
337  * We want to write num_data many timesteps/arrays.
338  * We need num_data many vtkDoubleArrays, so we need to allocate storage.
339  * Later we call the constructor with: dataArrays[idata]=vtkDoubleArray::New().
340  */
341  vtkDoubleArray **dataArrays;
342  dataArrays = T8_ALLOC (vtkDoubleArray *, num_data);
343 
344  long int point_id = 0; /* The id of the point in the points Object. */
345  const t8_gloidx_t offset
346  = grid_first_local_id (grid); /* offset to take the elements of the previous processes into account*/
347  t8_gloidx_t elem_id = offset;
348 
349  /* Check if we have to write ghosts on this process. */
350  bool do_ghosts = grid_do_ghosts (grid, write_ghosts);
351  /* Compute the number of cells on this process. */
352  t8_locidx_t num_cells = num_cells_to_write (grid, do_ghosts);
353 
354  int *cellTypes = T8_ALLOC (int, num_cells);
355  T8_ASSERT (cellTypes != NULL);
356 
357  /* Iterate over all trees and translate them. */
358  const t8_locidx_t num_local_trees = grid_local_num_trees (grid);
359  for (t8_locidx_t itree = 0; itree < num_local_trees; itree++) {
360  t8_grid_tree_to_vtk_cells (grid, unstructuredGrid, vtk_treeid, vtk_mpirank, vtk_level, vtk_element_id, cellArray,
361  points, cellTypes, num_local_trees, &elem_id, &point_id, offset, false, itree);
362  }
363  if (do_ghosts) {
364  /* Iterate over all ghost trees and translate them. */
365  const t8_locidx_t num_ghost_trees = grid_local_num_ghost_trees (grid);
366  for (t8_locidx_t itree_ghost = 0; itree_ghost < num_ghost_trees; itree_ghost++) {
367  t8_grid_tree_to_vtk_cells (grid, unstructuredGrid, vtk_treeid, vtk_mpirank, vtk_level, vtk_element_id,
368  cellArray, points, cellTypes, num_local_trees, &elem_id, &point_id, offset, true,
369  itree_ghost);
370  }
371  }
372 
373  /* Construct the unstructuredGrid. */
374  unstructuredGrid->SetPoints (points);
375  unstructuredGrid->SetCells (cellTypes, cellArray);
376 
377  if (this->write_treeid) {
378  vtk_treeid->SetName ("treeid");
379  unstructuredGrid->GetCellData ()->AddArray (vtk_treeid);
380  }
381  if (this->write_mpirank) {
382  vtk_mpirank->SetName ("mpirank");
383  unstructuredGrid->GetCellData ()->AddArray (vtk_mpirank);
384  }
385  if (this->write_level) {
386  vtk_level->SetName ("level");
387  unstructuredGrid->GetCellData ()->AddArray (vtk_level);
388  }
389  if (this->write_element_id) {
390  vtk_element_id->SetName ("element_id");
391  unstructuredGrid->GetCellData ()->AddArray (vtk_element_id);
392  }
393 
394  /* Write the user defined data fields. For that we iterate over the idata, set the name, the array and then give
395  * this data to the unstructured Grid Object.We differentiate between scalar and vector data.
396  */
397  const t8_locidx_t num_elements = grid_local_num_elements (grid);
398  for (int idata = 0; idata < num_data; idata++) {
399  dataArrays[idata] = vtkDoubleArray::New ();
400  const int num_components = this->data[idata].type == T8_VTK_SCALAR ? 1 : 3;
401  dataArrays[idata]->SetName (this->data[idata].description); /* Set the name of the array. */
402  dataArrays[idata]->SetNumberOfTuples (num_elements); /* We want number of tuples=number of elements. */
403  dataArrays[idata]->SetNumberOfComponents (num_components); /* Each tuples has 3 values. */
404  dataArrays[idata]->SetVoidArray (this->data[idata].data, num_elements * num_components, 1);
405  unstructuredGrid->GetCellData ()->AddArray (dataArrays[idata]);
406  }
407 
408  /* We have to free the allocated memory for the cellTypes Array and the other arrays we allocated memory for. */
409  for (int idata = 0; idata < num_data; idata++) {
410  dataArrays[idata]->Delete ();
411  }
412 
413  T8_FREE (cellTypes);
414  T8_FREE (dataArrays);
415  return;
416  }
417 #endif /* T8_WITH_VTK */

426  bool
427  write_vtk (const grid_t grid)
428  {
429 #if T8_WITH_VTK
430  T8_ASSERT (!fileprefix.empty ());
431 
432  /*
433  * Write file: First we construct the unstructured Grid that will store the points and elements. It requires
434  * information about the points(coordinates, stored in the points object) and the cells(cellTypes and which points
435  * belong to this cell).
436  */
437  vtkSmartPointer<vtkUnstructuredGrid> unstructuredGrid = vtkSmartPointer<vtkUnstructuredGrid>::New ();
438 
439  std::string mpifilename = fileprefix + std::string (".pvtu");
440 
441  vtkSmartPointer<vtkXMLPUnstructuredGridWriter> pwriterObj = vtkSmartPointer<vtkXMLPUnstructuredGridWriter>::New ();
442  t8_grid_to_vtkUnstructuredGrid (grid, unstructuredGrid);
443  /*
444  * Get/Set whether the appended data section is base64 encoded.
445  * If encoded, reading and writing will be slower, but the file
446  * will be fully valid XML and text-only.
447  * If not encoded, the XML specification will be violated,
448  * but reading and writing will be fast. The default is to do the encoding.
449  * Documentation: https://vtk.org/doc/release/5.0/html/a02260.html#z3560_2
450  */
451  pwriterObj->EncodeAppendedDataOff ();
452 
453  /* We set the filename of the pvtu file. The filenames of the vtu files
454  * are given based on the name of the pvtu file and the process number.
455  */
456  pwriterObj->SetFileName (mpifilename.c_str ());
457 
458 /*
459  * Since we want to write multiple files, the processes
460  * have to communicate. Therefore, we define the communicator
461  * vtk_comm and set it as the communicator.
462  * We have to set a controller for the pwriterObj,
463  * therefore we define the controller vtk_mpi_ctrl.
464  */
465 #if T8_ENABLE_MPI
466  vtkSmartPointer<vtkMPICommunicator> vtk_comm = vtkSmartPointer<vtkMPICommunicator>::New ();
467  vtkMPICommunicatorOpaqueComm vtk_opaque_comm (&comm);
468  vtk_comm->InitializeExternal (&vtk_opaque_comm);
469 
470  vtkSmartPointer<vtkMPIController> vtk_mpi_ctrl = vtkSmartPointer<vtkMPIController>::New ();
471  vtk_mpi_ctrl->SetCommunicator (vtk_comm);
472 
473  pwriterObj->SetController (vtk_mpi_ctrl);
474 #endif /* T8_ENABLE_MPI */
475  /*
476  * We set the number of pieces as the number of mpi processes,
477  * since we want to write a file for each process. We also
478  * need to define a Start and EndPiece for the current
479  * process. Then we can set the inputData for the writer:
480  * We want to write the unstructured Grid, update the writer
481  * and then write.
482  *
483  * Note: We could write more than one file per process here, if desired.
484  */
485  int mpisize;
486  int mpirank;
487  int mpiret = sc_MPI_Comm_size (comm, &mpisize);
488  SC_CHECK_MPI (mpiret);
489  mpiret = sc_MPI_Comm_rank (comm, &mpirank);
490  SC_CHECK_MPI (mpiret);
491 
492  pwriterObj->SetNumberOfPieces (mpisize);
493  pwriterObj->SetStartPiece (mpirank);
494  pwriterObj->SetEndPiece (mpirank);
495 
496  /* We set the input data and write the vtu files. */
497  pwriterObj->SetInputData (unstructuredGrid);
498  pwriterObj->Update ();
499  if (pwriterObj->Write ()) {
500  return true;
501  }
502  else {
503  t8_errorf ("Error when writing vtk file.\n");
504  }
505 
506  /* Return whether writing was successful */
507  return false;
508 
509 #else
510  t8_global_errorf ("Warning: t8code is not linked against vtk library. Vtk output will not be generated.\n");
511  t8_global_productionf ("Consider calling 't8_forest_write_vtk' or 't8_forest_vtk_write_file' instead.\n");
512  return false;
513 #endif
514  }

522  std::string fileprefix;
523  int num_data = 0;
524  t8_vtk_data_field_t *data = NULL;
525  sc_MPI_Comm comm;
