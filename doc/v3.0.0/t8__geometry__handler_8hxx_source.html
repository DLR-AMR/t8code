256  template <typename geometry_type>
257  inline geometry_type *
258  add_geometry (std::unique_ptr<t8_geometry> geom)
259  {
260  t8_debugf ("Registering geometry with name %s\n", geom->t8_geom_get_name ().c_str ());
261  const size_t hash = geom->t8_geom_get_hash ();
262  if (registered_geometries.find (hash) == registered_geometries.end ()) {
263  registered_geometries.emplace (hash, std::move (geom));
264  }
265  else {
266  t8_productionf ("WARNING: Did not register the geometry %s because it is already registered.\n"
267  "Geometries only need to be registered once per process.\n"
268  "If you are registering a new geometry it probably has the same name as another one.\n",
269  geom->t8_geom_get_name ().c_str ());
270  }
271  if (registered_geometries.size () == 1) {
272  active_geometry = registered_geometries.at (hash).get ();
273  }
274  return static_cast<geometry_type *> (registered_geometries.at (hash).get ());
275  }
