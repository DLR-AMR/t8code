584  {
585 #if T8_ENABLE_DEBUG
586  const int num_siblings = element_get_num_siblings (fam[0]);

588  T8_ASSERT (element_is_valid (fam[isib]));

598  element_get_parent ((const t8_element_t *) fam[childid], (t8_element_t *) &compare);

1120  static constexpr void
1121  element_set_linear_id (t8_element_t *elem, const t8_element_level level, t8_linearidx_t id) noexcept
1122  {
1123 
1124  t8_standalone_element<TEclass> *el = (t8_standalone_element<TEclass> *) elem;
1125 
1126  set_to_root ((t8_element_t *) el);
1127 
1128  /* There is only one element at level 0, so it must be root */
1129  if (level == 0) {
1130  T8_ASSERT (id == 0);
1131  return;
1132  }
1133 
1134  T8_ASSERT (id < (size_t) element_count_leaves (elem, level));
1135  T8_ASSERT (1 <= level && level <= T8_ELEMENT_MAXLEVEL[TEclass]);
1136  t8_standalone_element<TEclass> child;
1137 
1138  while (el->level < level) {
1139  /* Shortcut if we need the first descendant of the subtree*/
1140  if (id == 0) {
1141  element_get_first_descendant ((const t8_element_t *) el, (t8_element_t *) el, level);
1142  return;
1143  }
1144 
1145  t8_linearidx_t sum_descendants_of_children_before;
1146  t8_linearidx_t sum_descendants_of_children_until_current = 0;
1147  int childindex = -1;
1148 
1149  /* Find the first child id so that the sum of descendants of previous child and the own number of descendants is greater than id */
1150  do {
1151  /* Go to the next child */
1152  sum_descendants_of_children_before = sum_descendants_of_children_until_current;
1153  childindex++;
1154  T8_ASSERT (childindex < element_get_num_children ((const t8_element_t *) el));
1155 
1156  element_get_num_children ((const t8_element_t *) el);
1157 
1158  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) &child);
1159  const t8_linearidx_t num_descendants_of_child = element_count_leaves ((t8_element_t *) &child, level);
1160 
1161  /* Add number of descendant of current child to cumulative sum */
1162  sum_descendants_of_children_until_current = sum_descendants_of_children_before + num_descendants_of_child;
1163 
1164  } while (sum_descendants_of_children_until_current <= id);
1165 
1166  /* Replace el by child to go into next iteration at finer level*/
1167  element_get_child ((const t8_element_t *) el, childindex, (t8_element_t *) el);
1168  /* get id in subtree of child */
1169  id -= sum_descendants_of_children_before;
1170  }
1171  T8_ASSERT (id == 0);
1172  return;
1173  }

1182  element_get_linear_id (const t8_element_t *elem, const t8_element_level level) noexcept
1183  {
1184  T8_ASSERT (element_is_valid (elem));
1185  const t8_standalone_element<TEclass> *el = (const t8_standalone_element<TEclass> *) elem;
1186  t8_standalone_element<TEclass> ancestor;
1187 
1188  /* Determine the starting element for the iterative linear id computation. */
1189  if (level < el->level) {
1190  /* Throw away child ids up to the coarser level */
