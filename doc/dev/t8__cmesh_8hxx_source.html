71 constexpr uint64_t
72 t8_cmesh_get_first_element_of_process (const uint32_t process, const uint32_t mpisize,
73  const uint64_t global_num_elements)
74 {
75  T8_ASSERT (mpisize > 0);
76  T8_ASSERT (process <= mpisize);
77 
78  /* Cast everything into uint64_t */
79  const uint64_t process_64 = static_cast<uint64_t> (process);
80  const uint64_t mpisize_64 = static_cast<uint64_t> (mpisize);
81 
82  /* Split the uint64_t */
83  const uint64_t elem_over_size = global_num_elements / mpisize_64;
84  const uint64_t remainder_0 = global_num_elements % mpisize_64;
85 
86  const uint64_t proc_over_size = process_64 / mpisize_64;
87  const uint64_t remainder_1 = process_64 % mpisize_64;
88 
89  const uint64_t sum_0 = (elem_over_size * proc_over_size) * mpisize_64;
90  const uint64_t sum_1 = elem_over_size * (process_64 % mpisize_64);
91  const uint64_t sum_2 = proc_over_size * (global_num_elements % mpisize_64);
92  const uint64_t sum_3 = (remainder_0 * remainder_1) / mpisize_64;
93 
94  return (sum_0 + sum_1 + sum_2 + sum_3);
95 }
