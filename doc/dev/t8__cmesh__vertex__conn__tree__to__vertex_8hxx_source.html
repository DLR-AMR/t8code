153  t8_gloidx_t
154  get_global_vertex (const t8_cmesh_t cmesh, const t8_locidx_t local_tree, const int local_tree_vertex) const
155  {
156  T8_ASSERT (t8_cmesh_is_committed (cmesh));
157 
158  /* Verify that local_tree_vertex is in fact a local vertex of the tree */
159  /* Note: We only perform this check in debugging mode.
160  * In non-debugging mode, using a vertex index beyond the trees index allows
161  * for a potential attacker to gain access to memory possibly not owned by the caller.
162  * We do not check in non-debugging mode for (obvious) performance reasons. */
163  T8_ASSERT (0 <= local_tree_vertex);
164  const std::span<const t8_gloidx_t> global_vertices = get_global_vertices (cmesh, local_tree);
165  T8_ASSERT (global_vertices.size () > static_cast<size_t> (local_tree_vertex));
166  return global_vertices[local_tree_vertex];
167  }
