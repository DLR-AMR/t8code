240  t8_grid_element_to_vtk_cell (const grid_t grid, const t8_element_t *element, const t8_locidx_t itree,
241  const t8_gloidx_t offset, const int is_ghost, const int elem_id, long int *point_id,
242  int *cellTypes, vtkSmartPointer<vtkMergePoints> points,
243  vtkSmartPointer<vtkCellArray> cellArray,
244  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_treeid,
245  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_mpirank,
246  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_level,
247  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_element_id, bool mergePoints = true)
248  {
249  /* Get the shape of the current element and the respective shape of the vtk_cell. */
250  const t8_element_shape_t element_shape = grid_element_shape (grid, itree, element);
251  const int num_node = t8_get_number_of_vtk_nodes (element_shape, curved_flag);
252 
253  /* Create an array with the correct number of indices per cell */
254  vtkIdType *vecCellIds = new vtkIdType[num_node];
255 
256  /* Compute the coordinates of the element/tree. */
257  double *coordinates = T8_ALLOC (double, 3 * num_node);
258 
259  grid_element_to_coords (grid, itree, element, curved_flag, coordinates, num_node, element_shape);
260 
261  vtkIdType ptId = -1;
262  for (int ivertex = 0; ivertex < num_node; ivertex++, (*point_id)++) {
263  const size_t offset_3d = 3 * ivertex;
264 
265  /* Insert the point in the points array. */
266  double vtkCoords[3] = { coordinates[offset_3d], coordinates[offset_3d + 1], coordinates[offset_3d + 2] };
267  if (mergePoints) {
268  points->InsertUniquePoint (vtkCoords, ptId);
269  }
270  else {
271  ptId = points->InsertNextPoint (vtkCoords);
272  }
273  /* Add the returned point id to the cell ids*/
274  vecCellIds[ivertex] = ptId;
275  }
276  T8_FREE (coordinates);
277 
278  /* Fill the cell array. */
279  cellArray->InsertNextCell (num_node, vecCellIds);
280  delete[] vecCellIds;
281 
282  /* Write additional information if desired. */
283  if (curved_flag == 0) {
284  cellTypes[elem_id - offset] = t8_eclass_vtk_type[element_shape];
285  }
286  else {
287  cellTypes[elem_id - offset] = t8_curved_eclass_vtk_type[element_shape];
288  }
289  if (write_treeid == 1) {
290  const t8_gloidx_t gtree_id = tree_local_to_global_id (grid, itree);
291  if (is_ghost) {
292  vtk_treeid->InsertNextValue (-1);
293  }
294  else {
295  vtk_treeid->InsertNextValue (gtree_id);
296  }
297  }
298  if (write_mpirank == 1) {
299  int mpirank;
300  int mpiret = sc_MPI_Comm_rank (this->comm, &mpirank);
301  SC_CHECK_MPI (mpiret);
302  vtk_mpirank->InsertNextValue (mpirank);
303  }
304  if (write_level == 1) {
305  vtk_level->InsertNextValue (grid_element_level (grid, itree, element));
306  }
307  if (write_element_id == 1) {
308  vtk_element_id->InsertNextValue (elem_id);
309  }
310  }

348  void
349  t8_grid_to_vtkUnstructuredGrid (const grid_t grid, vtkSmartPointer<vtkUnstructuredGrid> unstructuredGrid)
350  {
351  T8_ASSERT (grid != NULL);
352 
353  vtkSmartPointer<vtkCellArray> cellArray = vtkSmartPointer<vtkCellArray>::New ();
354  vtkSmartPointer<vtkPoints> points_store = vtkSmartPointer<vtkPoints>::New ();
355  vtkSmartPointer<vtkMergePoints> points = vtkSmartPointer<vtkMergePoints>::New ();
356  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_treeid = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
357  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_mpirank = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
358  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_level = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
359  vtkSmartPointer<t8_vtk_gloidx_array_type_t> vtk_element_id = vtkSmartPointer<t8_vtk_gloidx_array_type_t>::New ();
360 
361  /*
362  * We need the dataArray for writing double valued user defined data in the vtu files.
363  * We want to write num_data many timesteps/arrays.
364  * We need num_data many vtkDoubleArrays, so we need to allocate storage.
365  * Later we call the constructor with: dataArrays[idata]=vtkDoubleArray::New().
366  */
367  vtkDoubleArray **dataArrays;
368  dataArrays = T8_ALLOC (vtkDoubleArray *, num_data);
369 
370  long int point_id = 0; /* The id of the point in the points Object. */
371  const t8_gloidx_t offset
372  = grid_first_local_id (grid); /* offset to take the elements of the previous processes into account*/
373  t8_gloidx_t elem_id = offset;
374 
375  /* Check if we have to write ghosts on this process. */
376  bool do_ghosts = grid_do_ghosts (grid, write_ghosts);
377 
378  /* Compute the number of cells on this process. */
379  t8_locidx_t num_cells = num_cells_to_write (grid, do_ghosts);
380 
381  int *cellTypes = T8_ALLOC (int, num_cells);
382  T8_ASSERT (cellTypes != NULL);
383 
384  /* Get the local bounds of the forest or cmesh */
385  double bounds[6];
386  grid_get_local_bounds (grid, bounds);
387 
388  /* Allocate VTK Memory for the arrays */
389  const int grid_dim = grid_get_dim (grid);
390  const int maximum_cell_size = curved_flag ? t8_curved_dim_max_nodes[grid_dim] : t8_dim_max_nodes[grid_dim];
391  cellArray->AllocateEstimate (num_cells, maximum_cell_size);
392  points_store->Allocate (num_cells * maximum_cell_size);
393  vtk_treeid->Allocate (num_cells);
394  vtk_mpirank->Allocate (num_cells);
395  vtk_level->Allocate (num_cells);
396  vtk_element_id->Allocate (num_cells);
397 
398  /* Initialie vtkMergePoints for insertion. It needs the bounds to create the internal acceleration structure */
399  points->InitPointInsertion (points_store, bounds);
400 
401  /* Iterate over all trees and translate them. */
402  const t8_locidx_t num_local_trees = grid_local_num_trees (grid);
403  for (t8_locidx_t itree = 0; itree < num_local_trees; itree++) {
404  t8_grid_tree_to_vtk_cells (grid, unstructuredGrid, vtk_treeid, vtk_mpirank, vtk_level, vtk_element_id, cellArray,
405  points, cellTypes, num_local_trees, &elem_id, &point_id, offset, false, itree);
406  }
407  if (do_ghosts) {
408  /* Iterate over all ghost trees and translate them. */
409  const t8_locidx_t num_ghost_trees = grid_local_num_ghost_trees (grid);
410  for (t8_locidx_t itree_ghost = 0; itree_ghost < num_ghost_trees; itree_ghost++) {
411  t8_grid_tree_to_vtk_cells (grid, unstructuredGrid, vtk_treeid, vtk_mpirank, vtk_level, vtk_element_id,
412  cellArray, points, cellTypes, num_local_trees, &elem_id, &point_id, offset, true,
413  itree_ghost);
414  }
415  }
416 
417  /* Construct the unstructuredGrid. */
418  unstructuredGrid->SetPoints (points->GetPoints ());
419  unstructuredGrid->SetCells (cellTypes, cellArray);
420 
421  if (this->write_treeid) {
422  vtk_treeid->SetName ("treeid");
423  unstructuredGrid->GetCellData ()->AddArray (vtk_treeid);
424  }
425  if (this->write_mpirank) {
426  vtk_mpirank->SetName ("mpirank");
427  unstructuredGrid->GetCellData ()->AddArray (vtk_mpirank);
428  }
429  if (this->write_level) {
430  vtk_level->SetName ("level");
431  unstructuredGrid->GetCellData ()->AddArray (vtk_level);
432  }
433  if (this->write_element_id) {
434  vtk_element_id->SetName ("element_id");
435  unstructuredGrid->GetCellData ()->AddArray (vtk_element_id);
436  }
437 
438  /* Write the user defined data fields. For that we iterate over the idata, set the name, the array and then give
439  * this data to the unstructured Grid Object.We differentiate between scalar and vector data.
440  */
441  const t8_locidx_t num_elements = grid_local_num_elements (grid);
442  for (int idata = 0; idata < num_data; idata++) {
443  dataArrays[idata] = vtkDoubleArray::New ();
444  const int num_components = this->data[idata].type == T8_VTK_SCALAR ? 1 : 3;
445  dataArrays[idata]->SetName (this->data[idata].description); /* Set the name of the array. */
446  dataArrays[idata]->SetNumberOfTuples (num_elements); /* We want number of tuples=number of elements. */
447  dataArrays[idata]->SetNumberOfComponents (num_components); /* Each tuples has 3 values. */
448  dataArrays[idata]->SetVoidArray (this->data[idata].data, num_elements * num_components, 1);
449  unstructuredGrid->GetCellData ()->AddArray (dataArrays[idata]);
450  }
451 
452  /* We have to free the allocated memory for the cellTypes Array and the other arrays we allocated memory for. */
453  for (int idata = 0; idata < num_data; idata++) {
454  dataArrays[idata]->Delete ();
455  }
456 
457  /* Release unused memory in the arrays */
458  unstructuredGrid->Squeeze ();
459 
460  T8_FREE (cellTypes);
461  T8_FREE (dataArrays);
462  return;
463  }

473  bool
474  write_vtk ([[maybe_unused]] const grid_t grid)
475  {
476 #if T8_ENABLE_VTK
477  T8_ASSERT (!fileprefix.empty ());
478 
479  /*
480  * Write file: First we construct the unstructured Grid that will store the points and elements. It requires
481  * information about the points(coordinates, stored in the points object) and the cells(cellTypes and which points
482  * belong to this cell).
483  */
484  vtkSmartPointer<vtkUnstructuredGrid> unstructuredGrid = vtkSmartPointer<vtkUnstructuredGrid>::New ();
485 
486  std::string mpifilename = fileprefix + std::string (".pvtu");
487 
488  vtkSmartPointer<vtkXMLPUnstructuredGridWriter> pwriterObj = vtkSmartPointer<vtkXMLPUnstructuredGridWriter>::New ();
489  t8_grid_to_vtkUnstructuredGrid (grid, unstructuredGrid);
490  /*
491  * Get/Set whether the appended data section is base64 encoded.
492  * If encoded, reading and writing will be slower, but the file
493  * will be fully valid XML and text-only.
494  * If not encoded, the XML specification will be violated,
495  * but reading and writing will be fast. The default is to do the encoding.
496  * Documentation: https://vtk.org/doc/release/5.0/html/a02260.html#z3560_2
497  */
498  pwriterObj->EncodeAppendedDataOff ();
499 
500  /* We set the filename of the pvtu file. The filenames of the vtu files
501  * are given based on the name of the pvtu file and the process number.
502  */
503  pwriterObj->SetFileName (mpifilename.c_str ());
504 
505 /*
506  * Since we want to write multiple files, the processes
507  * have to communicate. Therefore, we define the communicator
508  * vtk_comm and set it as the communicator.
509  * We have to set a controller for the pwriterObj,
510  * therefore we define the controller vtk_mpi_ctrl.
511  */
512 #if T8_ENABLE_MPI
513  vtkSmartPointer<vtkMPICommunicator> vtk_comm = vtkSmartPointer<vtkMPICommunicator>::New ();
514  vtkMPICommunicatorOpaqueComm vtk_opaque_comm (&comm);
515  vtk_comm->InitializeExternal (&vtk_opaque_comm);
516 
517  vtkSmartPointer<vtkMPIController> vtk_mpi_ctrl = vtkSmartPointer<vtkMPIController>::New ();
518  vtk_mpi_ctrl->SetCommunicator (vtk_comm);
519 
520  pwriterObj->SetController (vtk_mpi_ctrl);
521 #endif /* T8_ENABLE_MPI */
522  /*
523  * We set the number of pieces as the number of mpi processes,
524  * since we want to write a file for each process. We also
525  * need to define a Start and EndPiece for the current
526  * process. Then we can set the inputData for the writer:
527  * We want to write the unstructured Grid, update the writer
528  * and then write.
529  *
530  * Note: We could write more than one file per process here, if desired.
531  */
532  int mpisize;
533  int mpirank;
534  int mpiret = sc_MPI_Comm_size (comm, &mpisize);
535  SC_CHECK_MPI (mpiret);
536  mpiret = sc_MPI_Comm_rank (comm, &mpirank);
537  SC_CHECK_MPI (mpiret);
538 
539  pwriterObj->SetNumberOfPieces (mpisize);
540  pwriterObj->SetStartPiece (mpirank);
541  pwriterObj->SetEndPiece (mpirank);
542 
543  /* We set the input data and write the vtu files. */
544  pwriterObj->SetInputData (unstructuredGrid);
545  pwriterObj->Update ();
546  if (pwriterObj->Write ()) {
547  return true;
548  }
549  else {
550  t8_errorf ("Error when writing vtk file.\n");
551  }
552 
553  /* Return whether writing was successful */
554  return false;
555 
556 #else
557  t8_global_errorf ("Warning: t8code is not linked against vtk library. Vtk output will not be generated.\n");
558  t8_global_productionf ("Consider calling 't8_forest_write_vtk' or 't8_forest_vtk_write_file' instead.\n");
559  return false;
560 #endif
561  }
