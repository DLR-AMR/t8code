276  private:
283  template <typename geometry_type>
284  inline geometry_type *
285  add_geometry (std::unique_ptr<t8_geometry> geom)
286  {
287  t8_debugf ("Registering geometry with name %s\n", geom->t8_geom_get_name ().c_str ());
288  const t8_geometry_hash hash = geom->t8_geom_get_hash ();
289  if (registered_geometries.find (hash) == registered_geometries.end ()) {
290  registered_geometries.emplace (hash, std::move (geom));
291  }
292  else {
293  t8_productionf ("WARNING: Did not register the geometry %s because it is already registered.\n"
294  "Geometries only need to be registered once per process.\n"
295  "If you are registering a new geometry it probably has the same name as another one.\n",
296  geom->t8_geom_get_name ().c_str ());
297  }
298  if (registered_geometries.size () == 1) {
299  active_geometry = registered_geometries.at (hash).get ();
300  }
301  return static_cast<geometry_type *> (registered_geometries.at (hash).get ());
302  }
